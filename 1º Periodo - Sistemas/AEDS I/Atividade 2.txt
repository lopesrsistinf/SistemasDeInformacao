#include <stdio.h>
#include <stdlib.h>
#include <math.h>
int main()
{
int Num1, Num2, Resultado = 0;
char Operacao;
do{
printf("\nDigite um numero inteiro seguido de uma operação de adição, subtração, multiplicação, divisão, resto, raiz, potência");
printf(" \ne outro numero inteiro (Exemplo: 56 + 34) ou 0 0 0 para sair: ");
scanf("%d %c %d", &Num1, &Operacao, &Num2);
if (Num1 == 0) break;
if (Operacao == '+') Resultado = Num1 + Num2;
if (Operacao == '-') Resultado = Num1 - Num2;
if (Operacao == '*') Resultado = Num1 * Num2;
if (Operacao == '/') Resultado = Num1 / Num2;
if (Operacao == '%') Resultado = Num1 % Num2;
if (Operacao == '@') Resultado = sqrt(Num1);
if (Operacao == '#') Resultado = pow(Num1,(1.0/Num2));
if (Operacao == '^') Resultado = pow(Num1,Num2);
printf("\n\n\n %d %c %d = %d\n\n\n", Num1, Operacao, Num2, Resultado);
} while (1);
return 0;
}

    if (num1 > num2 && num1 > num3) maior = num1;
    else if (num2 > num1 && num2 > num3) maior = num2;

    else if (num2 < num1 && num2 < num3) menor = num2;
    else menor = num3;

int main(){
    int num,fatorial,i;
    printf("Digite um valor: ");
    scanf("%d",&fatorial);
    num = fatorial;
    for (i = fatorial-1; i>0; i--){
        fatorial *= i;
    }
    printf("O fatorial de %d! e: %d",num,fatorial);

}
