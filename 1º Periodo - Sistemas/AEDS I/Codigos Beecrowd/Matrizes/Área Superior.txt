#include <stdio.h>
 
int main(){
char O;
    double M[12][12],soma=0.0,media;
    scanf("%c",&O);
    for(int i=0; i < 12; i++){
        for(int j=0; j<12;j++){
            scanf("%lf",&M[i][j]);
        }
    }
    for(int i=0; i < 5; i++){
        for(int j=(i+1); j <= 10-i;j++){
                soma += M[i][j];
        }
    }
    media = soma/30.0;
    if(O == 'S'){
        printf("%.1lf\n",soma);
    } else if(O == 'M'){
        printf("%.1lf\n",media);
    }
    return 0;
}
