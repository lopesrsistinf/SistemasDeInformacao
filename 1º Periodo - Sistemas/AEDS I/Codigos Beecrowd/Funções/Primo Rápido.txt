#include <stdio.h>
#include <math.h>

void verificaPrimalidade(int X)
{
    if (X <= 1)
    {
        printf("Not Prime\n");
        return;
    }

    if (X == 2)
    {
        printf("Prime\n");
        return;
    }

    if (X % 2 == 0)
    {
        printf("Not Prime\n");
        return;
    }

    int limite = (int)sqrt(X);

    for (int i = 3; i <= limite; i += 2)
    {
        if (X % i == 0)
        {
            printf("Not Prime\n");
            return;
        }
    }

    printf("Prime\n");
}

int main()
{
    int N,X;

    if (scanf("%d", &N) != 1) return 1;

    for (int i = 0; i < N; i++)
    {

        if (scanf("%d", &X) != 1) continue;

        verificaPrimalidade(X);
    }

    return 0;
}
