#include <stdio.h>
#include <stdlib.h>
#include <math.h>
 
int main()
{
    int horaIni,minIni,horaFim,minFim,horas=0,minutos=0;
    scanf("%d %d %d %d",&horaIni,&minIni,&horaFim,&minFim);
    horas = horaFim - horaIni;
    minutos = minFim - minIni;
    if (horaIni == horaFim && minIni == minFim){
        horas = 24;
        minutos = 0;
    } 
    if (minutos < 0){
        minutos += 60;
        horas -= 1;
    } 
    if (horas < 0){
        horas += 24;
    }
    printf("O JOGO DUROU %d HORA(S) E %d MINUTO(S)\n",horas,minutos);
    return 0;
}
