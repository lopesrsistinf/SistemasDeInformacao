#include <stdio.h>
 
int main() {
    int codigo1,numPeca1,codigo2,numPeca2;
    double valor1,valor2,total;
    scanf("%d %d %lf\n",&codigo1,&numPeca1,&valor1);
    scanf("%d %d %lf\n",&codigo2,&numPeca2,&valor2);
    total = (numPeca1 * valor1) + (numPeca2 * valor2);
    printf("VALOR A PAGAR: R$ %.2lf\n",total);
    return 0;
}
