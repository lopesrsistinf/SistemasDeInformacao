#include <stdio.h>
 
int main()
{

    int N,horas = 0,minutos = 0,segundos = 0;
    scanf("%d",&N);
    for(int i=0; i<N;i++){
        if(segundos < 59){
            segundos++;
        } else if (minutos < 59){
            minutos++;
            segundos = 0;
        } else {
            horas++;
            minutos = 0;
            segundos = 0;
        }
    }
    printf("%d:%d:%d\n",horas,minutos,segundos);
    return 0;
}
