#include <stdio.h>
#include <math.h>
 
int main(){
    double V,D,area,altura,pi=3.14,R;
    while(scanf("%lf",&V) != EOF){
        scanf("%lf",&D);
        R = D/2;
        area = pi * pow(R,2);
        altura = V/area;
        printf("ALTURA = %.2lf\n",altura);
        printf("AREA = %.2lf\n",area);
    }
    return 0;
}
