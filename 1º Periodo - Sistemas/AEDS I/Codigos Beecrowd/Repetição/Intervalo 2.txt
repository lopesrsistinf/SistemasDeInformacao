#include <stdio.h>
 
int main(){
    int N,in=0,out=0;
    scanf("%d",&N);
    int X[N];
    for (int j=0;j<N;j++){
        scanf("%d",&X[j]);
    }
    for(int i=0;i<N;i++){
        if (X[i] >= 10 && X[i]<= 20){
            in++;
        } else {
            out++;
        }
    }
    printf("%d in\n%d out\n",in,out);
    return 0;
}
