#include <stdio.h>
 
int main() {

    int A,B,C,D,E,par=0,impar=0,positivo=0,negativo=0;
    scanf("%d %d %d %d %d",&A,&B,&C,&D,&E);
    int delicia[5] = {A,B,C,D,E};
    for (int i=0; i<5; i++){
        if(delicia[i]%2==0){
            par++;
        } else {
            impar++;
        }
        if(delicia[i]>0){
            positivo++;
        } else if(delicia[i]<0){
            negativo++;
        }
    }
    printf("%d valor(es) par(es)\n%d valor(es) impar(es)\n%d valor(es) positivo(s)\n%d valor(es) negativo(s)\n",par,impar,positivo,negativo);
    return 0;
}
