#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    int N, max;
    char linha[205];
    scanf("%d", &N);
    getchar();

    while (N--) {
        int contador[26] = {0};
        int index;
        int max = 0;
        char c;

        fgets(linha, sizeof(linha), stdin);

        for (int i = 0; linha[i] != '\0'; i++) {
            c = linha[i];
            index = -1;
            if (c >= 'A' && c <= 'Z') {
                index = c - 'A';
            } else if (c >= 'a' && c <= 'z') {
                index = c - 'a';
            }

            if (index != -1) {
                contador[index]++;
            }
        }

        for (int i = 0; i < 26; i++) {
            if (contador[i] > max) {
                max = contador[i];
            }
        }

        for (int i = 0; i < 26; i++) {
            if (contador[i] == max) {
                printf("%c", (char)(i + 'a'));
            }
        }
        printf("\n");
    }

    return 0;
}
