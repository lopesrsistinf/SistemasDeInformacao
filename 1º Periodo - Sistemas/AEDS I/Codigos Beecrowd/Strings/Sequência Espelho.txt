#include <stdio.h>
#include <stdlib.h>
#include <string.h>
 
int main() {
   int C;
    int a1, a2;
    scanf("%d", &C);

    while (C--) {
        scanf("%d %d", &a1, &a2);
        for (int i = a1; i <= a2; i++) {
            printf("%d", i);
        }
        for (int i = a2; i >= a1; i--) {
            int num = i;
            do {
                printf("%d", num % 10);
                num = num / 10;
            } while (num > 0);
        }
        printf("\n");
    }
    return 0;
}
