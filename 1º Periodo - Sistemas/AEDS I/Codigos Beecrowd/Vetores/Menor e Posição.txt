#include <stdio.h>
 
int main(){
    int N,menor = 0,posicao = 0;
    scanf("%d",&N);
    int X[N];
    for(int i = 0; i < N; i++){
        scanf("%d",&X[i]);
        if(i == 0){
            menor = X[0];
        }
        if(X[i] < menor){
            menor = X[i];
            posicao = i;
        }
    }
    printf("Menor valor: %d\nPosicao: %d\n",menor,posicao);
    return 0;
}
